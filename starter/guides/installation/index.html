Установка

Рекомендуется запуск с использованием Docker.

Server

version: "3"

services:
  starter-server:
    image: apisgarpun/apis:starter-server-20201111
    restart: always
    logging:
      driver: "none"
    ports:
      - "28341:28341"
      - "61616:61616"
      - "61613:61613"
    volumes:
      - ./activemq-data:/usr/app/activemq-data # папка для базы данных activemq
      - ./starter-server.conf.json:/usr/app/config/starter-server.conf.json # тут ваш конфиг

Agent

version: "3"

services:
  starter-agent:
    image: apisgarpun/apis:starter-agent-20201111
    network_mode: "host"
    restart: always
    logging:
      driver: none
    environment:
      DEBUG_MODE: 'true' # Параметр включает расширенное логирование агента для облегчения отладки
    volumes:
      - /var/run/docker.sock:/var/run/docker.sock
      - /:/rootfs
      - /usr/bin/docker:/usr/bin/docker
      - /var/run:/var/run
      - /sys:/sys
      - /var/lib/docker/:/var/lib/docker
      - ./agent-config.json:/usr/app/config/starter-agent.conf.json # тут ваш конфиг
